Senkronizasyon Kurulumu
Telefon ve PC arasında verilerinizi senkronize etmek için Google Apps Script kullanıyoruz. Bu ücretsiz bir çözümdür.
Adım 1
Google Sheets'te yeni bir tablo oluşturun.
Tablo URL'sinden spreadsheet ID'yi kopyalayın:
docs.google.com/spreadsheets/d/[BU KISIM]/edit
Adım 2
Google Sheets'te Uzantılar → Apps Script'i açın.
Aşağıdaki kodu yapıştırın ve "Deploy → New deployment → Web App" olarak yayınlayın.
"Execute as: Me" ve "Who has access: Anyone" seçin.
function doGet(e) { var ss = SpreadsheetApp.openById(e.parameter.id); var sh = ss.getSheets()[0]; var data = sh.getDataRange().getValues(); return ContentService .createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON); } function doPost(e) { var ss = SpreadsheetApp.openById(e.parameter.id); var sh = ss.getSheets()[0]; var row = JSON.parse(e.postData.contents); sh.appendRow(row); return ContentService .createTextOutput("ok"); } function doDelete(e) { var ss = SpreadsheetApp.openById(e.parameter.id); var sh = ss.getSheets()[0]; var id = e.parameter.rowId; var data = sh.getDataRange().getValues(); for (var i = 1; i < data.length; i++) { if (data[i][0] == id) { sh.deleteRow(i + 1); break; } } return ContentService .createTextOutput("ok"); }
